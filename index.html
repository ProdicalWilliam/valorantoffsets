static uint64_t ror64(uint64_t v, uint8_t n) {
    n &= 63;
    return n ? (v >> n) | (v << (64 - n)) : v;
}

static uint64_t gray_decode(uint64_t t) {
    return (t >> 1) ^ (((t >> 1) ^ (2 * t)) & 0xAAAAAAAAAAAAAAAAULL);
